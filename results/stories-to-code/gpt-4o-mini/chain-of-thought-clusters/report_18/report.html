from datetime import datetime
from typing import List, Dict, Optional

class Submission:
    def __init__(self, id: int, status: str, user_id: int, publish_status: str):
        self.id = id
        self.status = status
        self.user_id = user_id
        self.publish_status = publish_status
        self.created_at = datetime.now()
        self.updated_at = self.created_at

class User:
    def __init__(self, id: int, role: str):
        self.id = id
        self.role = role

class FABSFile:
    def __init__(self, id: int, agency_id: int, content: str):
        self.id = id
        self.agency_id = agency_id
        self.content = content
        self.created_at = datetime.now()

class Agency:
    def __init__(self, id: int, name: str):
        self.id = id
        self.name = name

class Database:
    def __init__(self):
        self.submissions: List[Submission] = []
        self.users: List[User] = []
        self.fabs_files: List[FABSFile] = []
        self.agencies: List[Agency] = []

    def add_submission(self, submission: Submission):
        self.submissions.append(submission)

    def add_user(self, user: User):
        self.users.append(user)

    def add_fabs_file(self, fabs_file: FABSFile):
        self.fabs_files.append(fabs_file)

    def add_agency(self, agency: Agency):
        self.agencies.append(agency)

    def get_submission_by_id(self, id: int) -> Optional[Submission]:
        for submission in self.submissions:
            if submission.id == id:
                return submission
        return None

    def get_user_by_id(self, id: int) -> Optional[User]:
        for user in self.users:
            if user.id == id:
                return user
        return None

    def get_agency_by_id(self, id: int) -> Optional[Agency]:
        for agency in self.agencies:
            if agency.id == id:
                return agency
        return None

class FABSSystem:
    def __init__(self, database: Database):
        self.database = database

    def create_submission(self, user_id: int) -> Submission:
        new_submission = Submission(id=len(self.database.submissions)+1, status='New', user_id=user_id, publish_status='Draft')
        self.database.add_submission(new_submission)
        return new_submission

    def update_submission_status(self, submission_id: int, status: str):
        submission = self.database.get_submission_by_id(submission_id)
        if submission:
            submission.status = status
            submission.updated_at = datetime.now()

    def publish_submission(self, submission_id: int):
        submission = self.database.get_submission_by_id(submission_id)
        if submission and submission.publish_status == 'Draft':
            submission.publish_status = 'Published'
            submission.updated_at = datetime.now()

    def deactivate_publish_button(self, submission_id: int):
        submission = self.database.get_submission_by_id(submission_id)
        if submission:
            print(f"Publish button deactivated for submission {submission_id}")

    def remove_last_four_zip(self, zip_code: str) -> str:
        if len(zip_code) > 4:
            return zip_code[:-4]
        return zip_code

    def generate_fabs_file(self, agency_id: int, content: str) -> FABSFile:
        new_file = FABSFile(id=len(self.database.fabs_files)+1, agency_id=agency_id, content=content)
        self.database.add_fabs_file(new_file)
        return new_file

    def log_submission_activity(self, submission_id: int, activity: str):
        submission = self.database.get_submission_by_id(submission_id)
        if submission:
            print(f"Activity for Submission {submission.id}: {activity}")
