class DFileGenerator:
    def __init__(self):
        self.cache = {}
        self.last_fpds_update = datetime.now()

    def generate_file(self, request_id: str, force: bool = False) -> Optional[str]:
        if not force and request_id in self.cache:
            logger.info(f"Returning cached D file for request {request_id}")
            return self.cache[request_id]
        
        if datetime.now() > self.last_fpds_update:
            file_data = f"D file data for {request_id} generated at {datetime.now()}"
            self.cache[request_id] = file_data
            logger.info(f"Generated new D file for request {request_id}")
            return file_data
        return None
