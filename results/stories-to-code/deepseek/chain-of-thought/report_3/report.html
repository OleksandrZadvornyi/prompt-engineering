class PublishStatus(Enum):
    DRAFT = "draft"
    PUBLISHED = "published"
    IN_PROGRESS = "in_progress"

class Submission:
    def __init__(self, submission_id: str, publish_status: PublishStatus, created_by: str, updated_at: datetime):
        self.submission_id = submission_id
        self.publish_status = publish_status
        self.created_by = created_by
        self.updated_at = updated_at
        self.validation_errors: List[str] = []
        self.file_content: Optional[Dict] = None

    def update_publish_status(self, new_status: PublishStatus) -> None:
        self.publish_status = new_status
        self.updated_at = datetime.now()
        logger.info(f"Submission {self.submission_id} status updated to {new_status}")

    def add_validation_error(self, error: str) -> None:
        self.validation_errors.append(error)
        logger.warning(f"Validation error added to submission {self.submission_id}: {error}")

    def clear_validation_errors(self) -> None:
        self.validation_errors = []
        logger.info(f"Validation errors cleared for submission {self.submission_id}")

class FABSProcessor:
    def __init__(self):
        self.submissions: Dict[str, Submission] = {}
        self.gtas_window = {"start": None, "end": None}

    def process_deletions(self, deletion_date: str) -> None:
        try:
            date_obj = datetime.strptime(deletion_date, "%m-%d-%Y")
            logger.info(f"Processing deletions for {date_obj.strftime('%Y-%m-%d')}")
        except ValueError:
            logger.error(f"Invalid date format {deletion_date}. Use MM-DD-YYYY")

    def update_fabs_submission(self, submission_id: str, new_status: PublishStatus) -> bool:
        if submission_id not in self.submissions:
            logger.error(f"Submission {submission_id} not found")
            return False
        self.submissions[submission_id].update_publish_status(new_status)
        return True

    def set_gtas_window(self, start: datetime, end: datetime) -> None:
        self.gtas_window = {"start": start, "end": end}
        logger.info(f"GTAS window set from {start} to {end}")

class ValidationRules:
    def __init__(self):
        self.rules = {
            "duns": self.validate_duns,
            "zip": self.validate_zip,
            "cfda": self.validate_cfda
        }

    def validate_duns(self, record: Dict) -> bool:
        # Check if DUNS is valid based on business rules
        return record.get("duns") and len(record["duns"]) == 9

    def validate_zip(self, record: Dict) -> bool:
        # Basic ZIP validation
        zip_code = str(record.get("zip", "")).strip()
        return len(zip_code) in (5, 10)  # 5 for ZIP, 10 for ZIP+4

    def validate_cfda(self, record: Dict) -> bool:
        # CFDA program number validation
        cfda = record.get("cfda", "")
        return bool(cfda) and cfda.isdigit() and len(cfda) <= 6

class AppDashboard:
    def __init__(self):
        self.submission_stats = {
            "pending": 0,
            "published": 0,
            "failed": 0
        }
        self.last_updated = datetime.now()

    def update_stats(self, submission: Submission) -> None:
        if submission.publish_status == PublishStatus.PUBLISHED:
            self.submission_stats["published"] += 1
        elif submission.publish_status == PublishStatus.DRAFT:
            self.submission_stats["pending"] += 1
        else:
            self.submission_stats["failed"] += 1
        self.last_updated = datetime.now()

    def get_stats(self) -> Dict:
        return {
            "stats": self.submission_stats,
            "last_updated": self.last_updated
        }
