class PublishStatus(Enum):
    DRAFT = "draft"
    PUBLISHED = "published"
    PENDING = "pending"
    FAILED = "failed"

class Submission:
    def __init__(self, submission_id: str, agency: str, publish_status: PublishStatus = PublishStatus.DRAFT):
        self.submission_id = submission_id
        self.agency = agency
        self.publish_status = publish_status
        self.created_at = datetime.now()
        self.updated_at = datetime.now()
        self.validation_errors: List[str] = []
        self.flex_fields: Dict[str, str] = {}

    def update_publish_status(self, new_status: PublishStatus):
        self.publish_status = new_status
        self.updated_at = datetime.now()
        logger.info(f"Submission {self.submission_id} status updated to {new_status.value}")

    def add_validation_error(self, error: str):
        self.validation_errors.append(error)
        logger.warning(f"Validation error added to submission {self.submission_id}: {error}")

    def get_validation_errors(self) -> List[str]:
        return self.validation_errors

    def add_flex_field(self, key: str, value: str):
        self.flex_fields[key] = value
        logger.debug(f"Flex field {key} added to submission {self.submission_id}")

class FABSSubmission(Submission):
    def __init__(self, submission_id: str, agency: str):
        super().__init__(submission_id, agency)
        self.file_path: Optional[str] = None
        self.is_historical: bool = False

    def upload_file(self, file_path: str):
        self.file_path = file_path
        logger.info(f"File uploaded for submission {self.submission_id}: {file_path}")

    def validate(self) -> bool:
        if not self.file_path:
            self.add_validation_error("No file uploaded")
            return False
        # Simulate validation logic
        if "invalid" in self.file_path:
            self.add_validation_error("Invalid file format")
            return False
        return True

    def publish(self) -> bool:
        if self.publish_status != PublishStatus.DRAFT:
            logger.error(f"Submission {self.submission_id} is not in draft status")
            return False
        self.update_publish_status(PublishStatus.PENDING)
        # Simulate publishing logic
        if self.validate():
            self.update_publish_status(PublishStatus.PUBLISHED)
            return True
        else:
            self.update_publish_status(PublishStatus.FAILED)
            return False
