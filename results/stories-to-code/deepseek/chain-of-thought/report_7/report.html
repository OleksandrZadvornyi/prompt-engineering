class UIDesignerTools:
    def __init__(self):
        self.testing_schedule = {}
        self.issues = []

    def schedule_user_testing(self, test_name: str, date: datetime.datetime, participants: List[str]):
        self.testing_schedule[test_name] = {
            "date": date,
            "participants": participants,
            "status": "scheduled"
        }
        logging.info(f"User testing '{test_name}' scheduled for {date} with {len(participants)} participants")

    def add_issue(self, issue: str, component: str, priority: str = "medium"):
        self.issues.append({
            "issue": issue,
            "component": component,
            "priority": priority,
            "status": "open",
            "created_at": datetime.datetime.now()
        })
        logging.info(f"New issue added for {component}: {issue}")

    def generate_testing_report(self, test_name: str) -> Optional[Dict]:
        if test_name not in self.testing_schedule:
            return None
        
        test = self.testing_schedule[test_name]
        report = {
            "test_name": test_name,
            "date": test["date"],
            "participants": test["participants"],
            "findings": [issue for issue in self.issues if issue["component"] == test_name],
            "status": test["status"]
        }
        logging.info(f"Generated testing report for '{test_name}'")
        return report
