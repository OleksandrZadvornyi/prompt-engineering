class ValidationRules:
    @staticmethod
    def validate_duns(duns: str, action_type: str, registration_date: datetime) -> bool:
        """Validate DUNS based on action type and registration date."""
        valid_action_types = {'B', 'C', 'D'}
        if action_type in valid_action_types and duns and len(duns) == 9:
            return registration_date is not None
        return False

    @staticmethod
    def validate_zip_code(zip_code: str, include_plus4: bool = True) -> bool:
        """Validate ZIP code, optionally including ZIP+4."""
        if include_plus4:
            return len(zip_code) in (5, 10) and zip_code.isdigit()
        return len(zip_code) == 5 and zip_code.isdigit()

class FileProcessor:
    def __init__(self):
        self.validation_rules = ValidationRules()

    def process_upload(self, file_path: str, file_type: str) -> Dict:
        """Process uploaded file and return validation results."""
        errors = []
        if not file_path.endswith(file_type):
            errors.append(f"Invalid file extension. Expected {file_type}")
        return {"file_path": file_path, "errors": errors, "is_valid": len(errors) == 0}

class SubmissionManager:
    def __init__(self):
        self.submissions = {}
        self.lock = False

    def submit(self, submission_id: str, data: Dict) -> Dict:
        """Submit data and prevent duplicate submissions."""
        if self.lock:
            return {"status": "error", "message": "Submission is currently locked"}
        if submission_id in self.submissions:
            return {"status": "error", "message": "Duplicate submission"}
        
        self.lock = True
        self.submissions[submission_id] = data
        self.lock = False
        return {"status": "success", "submission_id": submission_id}
